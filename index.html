Grade Calculator

Enter your Prelim grade to calculate the required Midterm and Final grades.

from pyscript import Element def calculate_grade(event): try: prelim = float(Element("prelim").element.value) except ValueError: Element("result").element.style.display = "block" Element("errorMsg").element.innerHTML = "Please enter a valid prelim grade." Element("prelimGrade").element.innerHTML = "" Element("midtermGrade").element.innerHTML = "" Element("passMessage").element.innerHTML = "" Element("honorMessage").element.innerHTML = "" return passing_grade = 75 honor_grade = 90 prelim_percent = 0.20 midterm_percent = 0.30 final_percent = 0.50 if prelim < 0 or prelim > 100: Element("result").element.style.display = "block" Element("errorMsg").element.innerHTML = "Please enter a valid prelim grade between 0 and 100." Element("prelimGrade").element.innerHTML = "" Element("midtermGrade").element.innerHTML = "" Element("passMessage").element.innerHTML = "" Element("honorMessage").element.innerHTML = "" return current_total = prelim * prelim_percent required_total = passing_grade - current_total if required_total > 0: required_midterm_and_final = required_total / (midterm_percent + final_percent) pass_message = "You have a chance to pass!" if required_midterm_and_final <= 90 else "It may be challenging to pass." else: required_midterm_and_final = 0 pass_message = "Your current grade is sufficient to pass!" if prelim >= honor_grade: honor_message = "You are on track for academic honors!" else: required_honor_total = honor_grade - current_total required_honor_midfinal = required_honor_total / (midterm_percent + final_percent) honor_message = f"To achieve academic honors, you need {required_honor_midfinal:.2f} for both midterm and final grades." Element("errorMsg").element.innerHTML = "" Element("prelimGrade").element.innerHTML = f"Prelim Grade: {prelim:.2f}" Element("midtermGrade").element.innerHTML = f"Required Midterm and Final Grades: {required_midterm_and_final:.2f}" Element("passMessage").element.innerHTML = pass_message Element("honorMessage").element.innerHTML = honor_message Element("result").element.style.display = "block" Element("calculate-btn").element.onclick = calculate_grade
